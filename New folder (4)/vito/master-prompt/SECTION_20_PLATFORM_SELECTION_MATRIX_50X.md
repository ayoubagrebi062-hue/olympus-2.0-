## 6.1 Development Cost Calculator

```typescript
// PLATFORM COST CALCULATOR

interface ProjectRequirements {
  complexity: 'simple' | 'medium' | 'complex';
  hasAuth: boolean;
  hasDatabase: boolean;
  hasPayments: boolean;
  hasMobile: boolean;
  monthlyUsers: number;
  teamSize: number;
}

const calculateCosts = (req: ProjectRequirements) => {
  // Platform costs (monthly)
  const platformCosts = {
    // AI Development
    ai: {
      free: 0,
      boltnewPro: 20,
      lovablePro: 50,
      cursorPro: 20,
      copilot: 10,
    },

    // Backend
    backend: {
      supabaseFree: 0,
      supabasePro: 25,
      firebaseFree: 0,
      // Firebase scales with usage
    },

    // Hosting
    hosting: {
      vercelFree: 0,
      vercelPro: 20,
      netlifyFree: 0,
      netlifyPro: 19,
    },
  };

  // Estimate based on complexity
  const estimates = {
    simple: {
      buildMessages: 50,      // Lovable messages
      buildTokens: 2_000_000, // Bolt tokens
      cursorHours: 10,
      totalWeeks: 1,
    },
    medium: {
      buildMessages: 150,
      buildTokens: 5_000_000,
      cursorHours: 30,
      totalWeeks: 3,
    },
    complex: {
      buildMessages: 400,
      buildTokens: 15_000_000,
      cursorHours: 80,
      totalWeeks: 8,
    },
  };

  const est = estimates[req.complexity];

  // Calculate one-time build cost
  const buildCost = {
    lovableApproach: Math.ceil(est.buildMessages / 500) * 50,  // $50 per 500 messages
    boltApproach: Math.ceil(est.buildTokens / 10_000_000) * 20, // $20 per 10M tokens
    cursorCost: Math.ceil(est.cursorHours / 160) * 20,         // $20 per ~160 hours
  };

  // Calculate monthly operational cost
  const monthlyCost = {
    hosting: req.monthlyUsers > 1000 ? 20 : 0,
    backend: req.monthlyUsers > 50000 ? 25 : 0,
    maintenance: 20, // Cursor for maintenance
  };

  return {
    buildCost,
    monthlyCost,
    firstYearTotal:
      Math.min(buildCost.lovableApproach, buildCost.boltApproach) +
      (monthlyCost.hosting + monthlyCost.backend + monthlyCost.maintenance) * 12,
  };
};

// Example calculations
const examples = {
  simpleAMVP: calculateCosts({
    complexity: 'simple',
    hasAuth: true,
    hasDatabase: true,
    hasPayments: false,
    hasMobile: false,
    monthlyUsers: 100,
    teamSize: 1,
  }),
  // Result: Build $50-100, Monthly $20, Year 1: ~$340

  mediumSaaS: calculateCosts({
    complexity: 'medium',
    hasAuth: true,
    hasDatabase: true,
    hasPayments: true,
    hasMobile: false,
    monthlyUsers: 5000,
    teamSize: 1,
  }),
  // Result: Build $150-200, Monthly $45, Year 1: ~$740
```

```typescript
// ROI CALCULATION FOR AI PLATFORM INVESTMENT

const calculateROI = (scenario: {
  traditionalCost: number;
  traditionalTime: number;  // hours
  aiPlatformCost: number;
  aiTimeInvestment: number; // hours
  hourlyRate: number;
}) => {
  const {
    traditionalCost,
    traditionalTime,
    aiPlatformCost,
    aiTimeInvestment,
    hourlyRate,
  } = scenario;

  // Total cost comparison
  const aiTotalCost = aiPlatformCost + (aiTimeInvestment * hourlyRate);
  const costSavings = traditionalCost - aiTotalCost;
  const costSavingsPercent = (costSavings / traditionalCost) * 100;

  // Time comparison
  const timeSaved = traditionalTime - aiTimeInvestment;
  const timeSavingsPercent = (timeSaved / traditionalTime) * 100;

  // ROI calculation
  const investment = aiPlatformCost;
  const returns = costSavings;
  const roi = ((returns - investment) / investment) * 100;

  return {
    traditionalTotal: traditionalCost,
    aiTotal: aiTotalCost,
    costSavings,
    costSavingsPercent: costSavingsPercent.toFixed(1),
    timeSaved,
    timeSavingsPercent: timeSavingsPercent.toFixed(1),
    roi: roi.toFixed(0),
    paybackPeriod: investment / (costSavings / 12), // months
  };
};
```
